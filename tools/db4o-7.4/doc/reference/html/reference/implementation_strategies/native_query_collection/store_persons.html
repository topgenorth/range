01Private Shared Sub StorePilotsAndTrainees()Sub StorePilotsAndTrainees()
02            File.Delete(Db4oFileName)
03            Dim container As IObjectContainer = Database()
04            If Not container Is Nothing Then
05                Try
06                    Dim pilot As Pilot
07                    Dim trainee As Trainee
08                    Dim i As Integer
09                    For i = 0 To ObjectCount - 1 Step i + 1
10                        pilot = New Pilot("Professional Pilot #" + i.ToString(), i)
11                        trainee = New Trainee("Trainee #" + i.ToString(), pilot)
12                        container.Set(trainee)
13                    Next
14                    container.Commit()
15                Catch ex As Db4oException
16                    System.Console.WriteLine("Db4o Exception: " + ex.Message)
17                Catch ex As Exception
18                    System.Console.WriteLine("System Exception: " + ex.Message)
19                Finally
20                    CloseDatabase()
21                End Try
22            End If
23        End Sub
